-- 1、查询 “c001” 课程比 “c002” 课程成绩高的所有学生的学号；
-- 理解：“c001” 课程最高分比 “c002” 课程最高分要高的学生的学号
SELECT
	s.sno AS 学号 
FROM
	student AS s 
WHERE
	( SELECT MAX ( sc.score ) FROM sc WHERE sc.sno = s.sno AND sc.cno = 'c001' ) > ( SELECT MAX ( sc.score ) FROM sc WHERE sc.sno = s.sno AND sc.cno = 'c002' );
-- 2、查询平均成绩大于 60 分的同学的学号和平均成绩；
-- 理解：不限学科，只看成绩
SELECT
	s.sno AS 学号,
	AVG ( sc.score ) AS 平均成绩 
FROM
	student AS s
	JOIN sc ON s.sno = sc.sno

	s.sno

	AVG ( sc.score ) >= 60;

	student AS s;
